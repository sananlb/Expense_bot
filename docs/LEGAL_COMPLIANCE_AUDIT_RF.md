from django.db import transaction
from expenses.models import Profile, Household
from typing import Tuple
import logging

logger = logging.getLogger(__name__)


@sync_to_async
def delete_user_profile(telegram_id: int) -> Tuple[bool, str]:
    """
    –ü–æ–ª–Ω–æ–µ —É–¥–∞–ª–µ–Ω–∏–µ –ø—Ä–æ—Ñ–∏–ª—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –∏ –≤—Å–µ—Ö —Å–≤—è–∑–∞–Ω–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö.

        - (False, "NOT_FOUND") ‚Äî –ø—Ä–æ—Ñ–∏–ª—å –Ω–µ –Ω–∞–π–¥–µ–Ω
        - (False, "ERROR") ‚Äî –æ—à–∏–±–∫–∞ —É–¥–∞–ª–µ–Ω–∏—è

            if not profile:

                return (False, "NOT_FOUND")

            profile_id = profile.id
            household_id = profile.household_id

            # –û–±–Ω—É–ª—è–µ–º creator –≤ household (–µ—Å–ª–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å —Å–æ–∑–¥–∞—Ç–µ–ª—å)

                f"profile_id={profile_id}, was_in_household={household_id}"

            return (True, "")
